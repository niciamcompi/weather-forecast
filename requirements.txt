"""
Modelltraining mit LightGBM (Multi-Horizon Forecasting).
MLflow Tracking für kostenlose Experiment-Verwaltung (lokal oder Dagshub).

Für jede Stadt + jeden Horizont wird ein eigenes Modell trainiert.
Alternativ: ein globales Modell mit Stadt-Dummies.
"""

import pandas as pd
import numpy as np
import lightgbm as lgb
import mlflow
import mlflow.lightgbm
from sklearn.metrics import mean_absolute_error, mean_squared_error
from sklearn.model_selection import TimeSeriesSplit
from pathlib import Path
import json
import logging
import pickle

log = logging.getLogger(__name__)
logging.basicConfig(level=logging.INFO, format="%(asctime)s %(levelname)s %(message)s")

PROCESSED_DIR = Path("data/processed")
MODELS_DIR = Path("models")
MODELS_DIR.mkdir(parents=True, exist_ok=True)

FORECAST_HORIZONS = [1, 3, 6, 12, 24, 48]

# ── LightGBM Hyperparameter ───────────────────────────────────────────────────
LGBM_PARAMS = {
    "objective":       "regression",
    "metric":          "mae",
    "n_estimators":    1000,
    "learning_rate":   0.05,
    "num_leaves":      127,
    "max_depth":       -1,
    "min_child_samples": 20,
    "feature_fraction": 0.8,
    "bagging_fraction": 0.8,
    "bagging_freq":    5,
    "reg_alpha":       0.1,
    "reg_lambda":      0.1,
    "n_jobs":          -1,
    "verbose":         -1,
}


def train_global_model(
    df: pd.DataFrame,
    feature_cols: list,
    target_col: str,
    horizon: int,
    experiment_name: str = "swiss-weather-forecast",
) -> lgb.LGBMRegressor:
    """
    Trainiert ein globales Modell über alle Städte für einen Horizont.
    Nutzt TimeSeriesSplit für korrektes Walk-Forward-Splitting.
    """
    y_col = f"{target_col}_target_{horizon}h"

    # Nur Zeilen mit gültigem Target
    mask = df[y_col].notna()
    X = df.loc[mask, feature_cols].copy()
    y = df.loc[mask, y_col].copy()

    log.info(f"Training: Horizont={horizon}h | Ziel={target_col} | n={len(X):,}")

    # TimeSeriesSplit
    tscv = TimeSeriesSplit(n_splits=5)

    val_maes, val_rmses = [], []
    best_model = None
    best_mae = float("inf")

    with mlflow.start_run(run_name=f"lgbm_{target_col}_{horizon}h"):
        mlflow.log_params({**LGBM_PARAMS, "horizon_h": horizon, "target": target_col})

        for fold, (train_idx, val_idx) in enumerate(tscv.split(X)):
            X_train, X_val = X.iloc[train_idx], X.iloc[val_idx]
            y_train, y_val = y.iloc[train_idx], y.iloc[val_idx]

            model = lgb.LGBMRegressor(**LGBM_PARAMS)
            model.fit(
                X_train, y_train,
                eval_set=[(X_val, y_val)],
                callbacks=[lgb.early_stopping(50), lgb.log_evaluation(period=0)],
            )

            preds = model.predict(X_val)
            mae  = mean_absolute_error(y_val, preds)
            rmse = np.sqrt(mean_squared_error(y_val, preds))

            val_maes.append(mae)
            val_rmses.append(rmse)
            log.info(f"  Fold {fold+1}: MAE={mae:.3f}, RMSE={rmse:.3f}")

            if mae < best_mae:
                best_mae = mae
                best_model = model

        avg_mae  = np.mean(val_maes)
        avg_rmse = np.mean(val_rmses)
        mlflow.log_metrics({"cv_mae": avg_mae, "cv_rmse": avg_rmse})
        log.info(f"  ⌀ CV: MAE={avg_mae:.3f}, RMSE={avg_rmse:.3f}")

        # Finales Modell auf allen Daten trainieren
        final_model = lgb.LGBMRegressor(**LGBM_PARAMS)
        final_model.fit(X, y, callbacks=[lgb.log_evaluation(period=0)])

        # Speichern
        model_path = MODELS_DIR / f"{target_col}_{horizon}h.pkl"
        with open(model_path, "wb") as f:
            pickle.dump(final_model, f)

        mlflow.lightgbm.log_model(final_model, artifact_path=f"lgbm_{target_col}_{horizon}h")
        log.info(f"  → Modell gespeichert: {model_path}")

    return final_model


def train_all_models(target_col: str = "temperature_2m", horizons: list = FORECAST_HORIZONS):
    """Trainiert Modelle für alle Horizonte."""
    from feature_engineering import get_feature_columns

    path = PROCESSED_DIR / "features.parquet"
    if not path.exists():
        raise FileNotFoundError(f"Feature-Matrix nicht gefunden: {path}")

    df = pd.read_parquet(path)
    feature_cols = get_feature_columns(df, horizons)

    # MLflow lokal oder auf DagsHub (gratis)
    mlflow.set_tracking_uri("mlruns")          # lokal
    # mlflow.set_tracking_uri("https://dagshub.com/USER/REPO.mlflow")  # gratis Cloud
    mlflow.set_experiment("swiss-weather-forecast")

    models = {}
    for h in horizons:
        model = train_global_model(df, feature_cols, target_col, h)
        models[h] = model
        log.info(f"✓ Horizont {h}h abgeschlossen")

    # Feature-Liste speichern (für Inference)
    meta = {"feature_cols": feature_cols, "target_col": target_col, "horizons": horizons}
    with open(MODELS_DIR / "model_meta.json", "w") as f:
        json.dump(meta, f, indent=2)

    log.info("Alle Modelle trainiert.")
    return models


if __name__ == "__main__":
    train_all_models(target_col="temperature_2m")
